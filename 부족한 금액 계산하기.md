fun solution(price: Int, money: Int, count: Int): Long {
    var answer: Long = money.toLong()
    var i = 1
    while (i <= count){
        answer -= price*i
        i++
    }
    if (answer >= 0) answer = 0
    else answer = -answer
    return answer
}

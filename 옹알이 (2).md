fun solution(babbling: Array<String>): Int {
    val say = arrayOf("aya","ye", "woo", "ma")
    var count = 0
    babbling.forEach {
        var a = it
        var lastSay = ""
        var valid = true

        while (a.isNotEmpty() && valid) {
            valid = false
            for (sound in say) {
                if (a.startsWith(sound) && sound != lastSay) {
                    a = a.removePrefix(sound)
                    lastSay = sound
                    valid = true
                    break
                }
            }
        }

        if (a.isEmpty()) {
            count++
        }
    }

    return count
}
